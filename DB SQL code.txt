CREATE TABLE  "USERS" 
   (	"ID" NUMBER, 
	"NAME" VARCHAR2(100) NOT NULL ENABLE, 
	"NUMROOM" NUMBER NOT NULL ENABLE, 
	"NUMNIGHTS" NUMBER NOT NULL ENABLE, 
	 CONSTRAINT "USERS_PK" PRIMARY KEY ("ID") ENABLE
   ) ;

CREATE OR REPLACE TRIGGER  "BI_USERS" 
  before insert on "USERS"               
  for each row  
begin   
  if :NEW."ID" is null then 
    select "USERS_SEQ".nextval into :NEW."ID" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_USERS" ENABLE;

CREATE TABLE  "EMPLOYEE"

begin   
  if :NEW."EMAIL" is null then 
    select "EMPLOYEE_SEQ".nextval into :NEW."EMAIL" from dual; 
  end if; 
end; 

/
ALTER TRIGGER  "BI_EMPLOYEE" ENABLE;
